show databases; -_-_-> Muestra las bases de datos que hay en el localhost

use nombre_bd -_-_->  Para empezar a trabajar en una base de datos

create table Nombre_Tabla(Id, datos que se quieran agregar);  -_-_-> Crear tabla

show Tables;  -_-_-> Muestra las tablas que hay en una base de datos

alter table Nombre_Tabla add primary key(Id Principal) -_-_-> Agregar llave primaria

alter table Nombre_Tabla add foreign key (Id Llave foranea) references Nombre_Tabla_Fk (Id_PK_FK); -_-_-> Agregar llaves foraneas, relacionar tablas

Drop table Nombre_Tabla; -_-_-> Eliminar una tabla por completo

Drop nombre_BD; -_-_-> Eliminar una base de datos (No intentar en casa)
